```ts
import { CanActivate, ExecutionContext, Injectable, UnauthorizedException } from '@nestjs/common';

@Injectable()
export class JwtAuthGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const request = context.switchToHttp().getRequest<{ user?: { id?: number } }>();
    if (!Number.isInteger(request.user?.id) || (request.user?.id ?? 0) <= 0) {
      throw new UnauthorizedException('Authentication required');
    }
    return true;
  }
}
```

@Controller('finance/wallet')

export class UserWalletController {

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> wallet history API | Authenticated user can send arbitrary query params and bearer/cookie credentials. |
| API -> Prisma | Finance service reads sensitive user financial ledger rows. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-54-01 | Elevation of Privilege | `GET /finance/wallet/transactions` | mitigate | Enforce `JwtAuthGuard`; derive user id only from `req.user.id`; do not accept userId query/body input. |
| T-54-02 | Information Disclosure | transaction projection | mitigate | Select explicit safe fields only; never return `rawBody` or other users' rows. |
| T-54-03 | Denial of Service | page/pageSize params | mitigate | Clamp page/pageSize and cap pageSize at 50. |
| T-54-04 | Tampering | query params | mitigate | Numeric normalization only; no dynamic sort/search in this phase. |
</threat_model>
